@code {
    private IList<SchemaObjectColumnDefinition> selectedColumns = new List<SchemaObjectColumnDefinition>();

    [Parameter]
    public string? ViewDisplayName { get; set; }

    [Parameter]
    public IReadOnlyList<SchemaObjectColumnDefinition> SavedColumns { get; set; } = Array.Empty<SchemaObjectColumnDefinition>();

    [Parameter]
    public bool CanEditColumns { get; set; }

    [Parameter]
    public bool IsBusy { get; set; }

    [Parameter]
    public EventCallback<SchemaObjectColumnDefinition?> EditColumnRequested { get; set; }

    private string ColumnsTitle =>
        string.IsNullOrWhiteSpace(ViewDisplayName)
            ? "Columns"
            : $"{ViewDisplayName} - Columns";

    private SchemaObjectColumnDefinition? SelectedColumn => selectedColumns.FirstOrDefault();

    protected override void OnParametersSet()
    {
        if (SelectedColumn != null &&
            !SavedColumns.Any(column => column.SchemaObjectColumnId == SelectedColumn.SchemaObjectColumnId))
        {
            selectedColumns = new List<SchemaObjectColumnDefinition>();
        }
    }

    private void HandleRowSelect(SchemaObjectColumnDefinition? column)
    {
        selectedColumns = column == null
            ? new List<SchemaObjectColumnDefinition>()
            : new List<SchemaObjectColumnDefinition> { column };
    }

    private Task HandleEditSelectedAsync()
    {
        return EditColumnRequested.InvokeAsync(SelectedColumn);
    }

    private RenderFragment GridHeader(string propertyName) => builder =>
    {
        var displayName = ReflectionUtils.GetDisplayName(typeof(SchemaObjectColumnDefinition), propertyName);
        var description = ReflectionUtils.GetDescription(typeof(SchemaObjectColumnDefinition), propertyName);

        builder.OpenElement(0, "div");
        builder.AddAttribute(1, "class", "manage-views-columns-grid-header");
        builder.AddContent(2, displayName);

        if (!string.IsNullOrWhiteSpace(description))
        {
            builder.OpenComponent<RadzenIcon>(3);
            builder.AddAttribute(4, "Icon", "help_outline");
            builder.AddAttribute(5, "MouseEnter", EventCallback.Factory.Create<ElementReference>(this, args => TooltipService.Open(args, description)));
            builder.CloseComponent();
        }

        builder.CloseElement();
    };
}
